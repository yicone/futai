CREATE PROCEDURE [dbo].[search_SearchByType]
(
      @MType varchar(20) = NULL,
      @SType varchar(50)= NULL,
      @Hotted bit = NULL
)
AS
BEGIN
SELECT  *
        FROM dbo.Product
        WHERE MainType=@MType AND SubType=ISNULL(@SType,SubType) AND Hotted = ISNULL(@Hotted,Hotted)
        
END



















GO
